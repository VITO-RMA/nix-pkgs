cmake_minimum_required(VERSION 3.16)
project(Qwt VERSION 6.3.0 LANGUAGES CXX)

set(CMAKE_AUTOMOC ON)

option(QWT_BUILD_EXAMPLES "Build examples" OFF)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets Concurrent PrintSupport)
find_package(Qt6 OPTIONAL_COMPONENTS OpenGL OpenGLWidgets Svg)

file(GLOB QWT_SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp)
file(GLOB QWT_HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/src/*.h)

# qwt_plot_glcanvas is Qt5-only (uses removed QGLWidget)
list(FILTER QWT_SOURCES EXCLUDE REGEX "qwt_plot_glcanvas")
list(FILTER QWT_HEADERS EXCLUDE REGEX "qwt_plot_glcanvas")

if(NOT TARGET Qt6::Svg)
    list(FILTER QWT_SOURCES EXCLUDE REGEX "qwt_plot_svgitem")
    list(FILTER QWT_HEADERS EXCLUDE REGEX "qwt_plot_svgitem")
endif()

if(NOT TARGET Qt6::OpenGLWidgets)
    list(FILTER QWT_SOURCES EXCLUDE REGEX "qwt_plot_opengl_canvas")
    list(FILTER QWT_HEADERS EXCLUDE REGEX "qwt_plot_opengl_canvas")
endif()

add_library(qwt ${QWT_SOURCES} ${QWT_HEADERS})

target_include_directories(qwt PUBLIC
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/src>
    $<INSTALL_INTERFACE:include/qwt>
)

target_link_libraries(qwt PUBLIC
    Qt6::Core Qt6::Gui Qt6::Widgets Qt6::Concurrent Qt6::PrintSupport
)

if(TARGET Qt6::OpenGLWidgets)
    target_link_libraries(qwt PUBLIC Qt6::OpenGLWidgets)
else()
    target_compile_definitions(qwt PRIVATE QWT_NO_OPENGL)
endif()

if(TARGET Qt6::Svg)
    target_link_libraries(qwt PUBLIC Qt6::Svg)
else()
    target_compile_definitions(qwt PRIVATE QWT_NO_SVG)
endif()

if(BUILD_SHARED_LIBS)
    target_compile_definitions(qwt PRIVATE QWT_DLL QWT_MAKEDLL)
    target_compile_definitions(qwt INTERFACE QWT_DLL)
endif()

set_target_properties(qwt PROPERTIES
    VERSION ${PROJECT_VERSION}
    SOVERSION ${PROJECT_VERSION_MAJOR}
)

include(GNUInstallDirs)

install(TARGETS qwt EXPORT QwtTargets
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)

install(FILES ${QWT_HEADERS}
    DESTINATION ${CMAKE_INSTALL_INCLUDEDIR}/qwt
)

install(EXPORT QwtTargets
    FILE QwtConfig.cmake
    NAMESPACE Qwt::
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/Qwt
)

include(CMakePackageConfigHelpers)
write_basic_package_version_file(
    "${CMAKE_CURRENT_BINARY_DIR}/QwtConfigVersion.cmake"
    VERSION ${PROJECT_VERSION}
    COMPATIBILITY SameMajorVersion
)
install(FILES "${CMAKE_CURRENT_BINARY_DIR}/QwtConfigVersion.cmake"
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/cmake/Qwt
)
